#Land Transfers from Britain to independent Canada
country_event = {
	id = 1234121

	trigger = {
		tag = ENG
		war = no
		CAN = { war = no }
		any_owned_province = {
			region = USA_1 #Alaska
			controlled_by = THIS
			any_neighbor_province = { owned_by = CAN }
		}
	}

	fire_only_once = yes
	
	mean_time_to_happen =  {
		months = 12
	}

	title = "EVTNAME1234121"
	desc = "EVTDESC1234121"

	option = {
		name = "EVTOPTA1234121"
		USA_1 = {
			remove_core = THIS
			add_core = CAN
		}
		any_owned = {
			limit = { region = USA_1 }
			secede_province = CAN
		}
		random_country = {
			limit = {
				is_sphere_leader_of = CAN
				NOT = { tag = THIS }
			}
			diplomatic_influence = { who = CAN value = -100 }
		}
		diplomatic_influence = { who = CAN value = 100 }
		relation = { who = CAN value = 50 }
		ai_chance = { factor = 90 }

	}
	option = {
		name = "EVTOPTB1234121"
		ai_chance = { factor = 10 }
	}
}

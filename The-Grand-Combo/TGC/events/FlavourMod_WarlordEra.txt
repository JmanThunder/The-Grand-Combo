#Xuantong Emperor (Puyi)
country_event = {
	id = 999961
	title = "EVTNAME999961"
	desc = "EVTDESC999961"
	picture = "puyi_child"
	fire_only_once = yes
	
	trigger = {
		tag = QNG
		year = 1906
		owns = 1616 #Peking
		exists = yes
		OR = {
			government = absolute_monarchy
			government = prussian_constitutionalism
			government = hms_government
		}
		NOT = { has_global_flag = xuantong_emperor }
	}
	
	mean_time_to_happen = {
		months = 48
		
		modifier = {
			factor = 0.5
			civilized = yes
		}
		
		modifier = {
			factor = 0.5
			LEG = { exists = yes }
		}
		
		modifier = {
			factor = 0.1
			year = 1908
		}
	}
	
	option = {
		name = "EVTOPTA999961"
		plurality = -2
		any_pop = {
			limit = {
				is_primary_culture = yes
			}
			consciousness = -1
			militancy = -1
		}
		scaled_consciousness = {
			ideology = liberal
			factor = 3
		}
		scaled_militancy = {
			ideology = liberal
			factor = 3
		}
		set_global_flag = xuantong_emperor
	}		
}

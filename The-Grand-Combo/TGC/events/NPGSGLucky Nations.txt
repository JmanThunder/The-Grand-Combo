#Lucky Nation 1
country_event = {

	id = 66699

	trigger = {
		civilized = yes
		ai = yes
		total_pops = 230000
		NOT = { is_greater_power = yes }
		year = 1836
		has_global_flag = lucky_nations_on
		NOT = {
			has_global_flag = lucky_one
		}
	}

	fire_only_once = yes

	mean_time_to_happen =  {
		months = 12

		modifier = {
			factor = 1
			year = 1836
		}
	}

	title = "EVTNAME32420"
	desc = "EVTDESC32420"
	picture = "Revolution"

	option = {
		name = "EVTOPTA32420"
		set_global_flag = lucky_one
		add_country_modifier = {
			name = lucky_nation
			duration = -1
		}
	}
}
